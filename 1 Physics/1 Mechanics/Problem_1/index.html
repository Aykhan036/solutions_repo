def max_height(v0, theta_deg, g=9.81): theta_rad = np.radians(theta_deg) return (v02 * np.sin(theta_rad)2) / (2 * g)

def plot_height_vs_angle(v0, g=9.81): angles = np.linspace(0, 90, 100) heights = max_height(v0, angles, g)

plt.figure()
plt.plot(angles, heights)
plt.title(f"Max Height vs. Angle (v0 = {v0} m/s)")
plt.xlabel("Angle of Projection (degrees)")
plt.ylabel("Maximum Height (m)")
plt.grid(True)
plt.savefig("height_vs_angle.png")
plt.show()
